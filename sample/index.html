	// 実装

	// 初期化の各オプション。各値なしだとデフォルト値が適用される
	// speed : コマ送りの速度（フレームレート）
	// divWidth : スプライトシートコマの横幅
	// divHeight : スプライトシートコマの高さ
	// xValue : 横方向のコマ枚数
	// yValue : 縦方向のコマ枚数
	// frameValue : 全部何コマあるか

	var sss = new SimpleSpriteSheet;

	/*
	* 動かしたいスプライトシート分、addItemする
	* @param {String} DivのID
	* @param {String} 画像のパス
	*/
	sss.init({
		speed 		: 160,
		divWidth 	: 120,
		divHeight 	: 370,
		xValue 		: 10,
		yValue 		: 3
	});

	// 最大5個まで追加できる
	sss.addItem("img_1" , "img/sample.jpg");
	sss.addItem("img_2" , "img/test_1.jpg");
	sss.addItem("img_3" , "img/test_5.jpg");
	sss.addItem("img_4" , "img/test_4.jpg");
	sss.addItem("img_5" , "img/test_5.jpg");

	// start SpriteSheet
	sss.play();

	// stop SpriteSheet
	sss.stop();
